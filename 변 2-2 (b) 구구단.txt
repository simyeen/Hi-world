주요 내용 :
처음에 i, j, num 3개의 변수를 선언해준다.
단 이때 num은 한 줄에 나타내고 싶은 단의 수를 나타내므로 값을 입력받는다.
그리고 swich문이 입력받은 num값을 판단한다. 출력시 두개의 for문이 각각 행(i)과 열(j)을 구축한다.
그 후 입력받은 값만큼 한 줄에 num만큼 단이 출력된다 

----------------------------------------------------------------*/

#include <stdio.h>

int main()
{
	int num;       // 한 줄에 나타내고 싶은 단의 수 선언
	int i, j;      //행과 열을 나타낼 변수 선언.

	printf("한 줄에 나타내고 싶은 단의 수를 입력하시오: (단 1에서 4까지의 수만 입력하시오.)");
	scanf("%d", &num);

	switch (num)
	{
	case 1:                           // 1입력시 1줄에 1단씩 출력한다.
		for (i = 2; i < 10; i++)
		{
			for (j = 1; j < 10; j++)
			{
				printf("%3d x %3d = %3d", i, j, i*j);
				printf("\n");
			}
			printf("\n");
		}
		break;

	case 2:                         //2 입력시 2줄에 2단씩 출력한다.
		for (i = 2; i < 10; i++)
		{
			for (j = 1; j < 10; j++)
			{
				printf("%3d x %3d = %3d  ", i, j, i*j);            // 2단 첫 줄 출력.
				printf("%3d x %3d = %3d  ", i + 1, j, (i + 1)*j);  // 3단 첫 줄 출력.
				printf("\n");
			}

			i++;                                                   // 제일 처음 for문의 i++와 52라인 덕분에 +2가 되어서 2단, 3단후 바로 4단, 5단을 출력하게 한다.
			printf("\n");                                          // 단 끼리 구분하기 위한 \n이다.
		}

		break;

	case 3:                        //3입력시 1줄에 3단씩 출력한다.
		for (i = 2; i < 10; i++)
		{
			for (j = 1; j < 10; j++)
			{
				printf("%3d x %3d = %3d  ", i, j, i*j);
				printf("%3d x %3d = %3d  ", i + 1, j, (i + 1)*j);
				if (i < 8)                                            // i=8때 10단이 출력되므로 조건식을 걸어준다.
					printf("%3d x %3d = %3d  ", i + 2, j, (i + 2)*j);
				printf("\n");

			}

			i += 2;
			printf("\n");
		}

		break;
	case 4:
		for (i = 2; i < 10; i++)
		{
			for (j = 1; j < 10; j++)
			{
				printf("%3d x %3d = %3d  ", i, j, i*j);
				printf("%3d x %3d = %3d  ", i + 1, j, (i + 1)*j);
				printf("%3d x %3d = %3d  ", i + 2, j, (i + 2)*j);
				printf("%3d x %3d = %3d  ", i + 3, j, (i + 3)*j);
				printf("\n");
			}
			i += 3;

			printf("\n");
		}
		break;
	default:

		printf("1에서 4까지의 수만 입력하시오.");
		break;

	}

	return 0;
}
