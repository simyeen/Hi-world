void make_answer(int computer[]) {
	int i;

	srand((unsigned)time(NULL));
	for (i = 0; i < 3; i++)
		computer[i] = (rand() % 9) + 1;       //첫 번째 수에는 1이 나올수 없다.
	for (i = 0; i<3; i++) {                   //서로 중복되는 수가 나올수가 없음.
		if (computer[0] == computer[1])
			computer[1] = (rand() % 9);
		else if (computer[1] == computer[2])
			computer[2] = (rand() % 9);
		else if (computer[2] == computer[0])
			computer[2] = (rand() % 9);
		else if (computer[1] != computer[2] && computer[0] != computer[1] && computer[0] != computer[2])
			break;
	}
	return computer;
}
